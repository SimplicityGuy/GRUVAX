      },
      ...
    ],
    "total":    int,
    "offset":   M,
    "limit":    N,
    "has_more": bool                  # terminal signal — false on last page
  }

  403 — missing collection:read scope
  429 — Retry-After: <seconds> header + X-RateLimit-* (rate limits: 60/min + 600/hour per token)
