gruvax.units columns: id SMALLSERIAL PK, display_name TEXT, rows SMALLINT, cols SMALLINT,
  ordering SMALLINT, created_at TIMESTAMPTZ, updated_at TIMESTAMPTZ.

gruvax.cube_boundaries columns: unit_id SMALLINT FK→units(id), row SMALLINT, col SMALLINT,
  first_label TEXT NULL, first_catalog TEXT NULL, last_label TEXT NULL, last_catalog TEXT NULL,
  is_empty BOOLEAN NOT NULL DEFAULT FALSE, updated_at TIMESTAMPTZ.
  PK (unit_id, row, col). CHECK empty_or_complete (is_empty OR all four bound columns NOT NULL).

  collection_item_id, release_id, title, label, catalog_number, format, year, fts_vector,
  primary_artist (= artists.name), synced_at (= collection_items.updated_at).
  View body references UNQUALIFIED table names (collection_items, releases, artists) so the
  connection's search_path selects gruvax_dev (dev) or discogsography (prod).
